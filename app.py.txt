    st.stop()

st.markdown('<div class="poder-badge">Buscando en: Asamblea Legislativa</div>', unsafe_allow_html=True)

@st.cache_resource
def cargar_chat():
    db  = chromadb.PersistentClient(path="./chroma_db")
    col = db.get_collection("transparencia_cr")
    vs  = ChromaVectorStore(chroma_collection=col)
    idx = VectorStoreIndex.from_vector_store(vs)
    return idx.as_chat_engine(chat_mode="condense_plus_context", verbose=False)

engine = cargar_chat()

if "msgs" not in st.session_state:
    st.session_state.msgs = [{
        "role": "assistant",
        "content": "¡Hola! Soy el asistente de TransparenciaCR. Puedo buscar en las sesiones de la Asamblea Legislativa.\n\n**Probá preguntas como:**\n- ¿Qué dijo el diputado X sobre el salario mínimo?\n- ¿Se discutió la ley de pensiones en 2025?\n- ¿Qué temas abordó la comisión de hacienda este año?"
    }]

for m in st.session_state.msgs:
    with st.chat_message(m["role"]):
        st.markdown(m["content"])

if prompt := st.chat_input("Escribí tu pregunta sobre cualquier sesión legislativa..."):
    st.session_state.msgs.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.markdown(prompt)
    with st.chat_message("assistant"):
        with st.spinner("Buscando en las sesiones..."):
            resp = engine.chat(prompt)
            st.markdown(resp.response)
            if resp.source_nodes:
                with st.expander("📎 Ver fuentes"):
                    for n in resp.source_nodes[:3]:
                        url   = n.metadata.get("url", "")
                        tit   = n.metadata.get("titulo", "Sesión")
                        fecha = n.metadata.get("fecha", "")
                        st.markdown(f"• [{tit}]({url}) — {fecha}")
    st.session_state.msgs.append({"role": "assistant", "content": resp.response})
